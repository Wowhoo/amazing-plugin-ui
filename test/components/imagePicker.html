<script>
    $(document).ready(function() {
        $('.imagePicker').asImagePicker({
            onSelect: function(){
                this.set({
                    image: '../../images/nvnv.png'
                });
            },
            disabled: true
        });

        $('.test_val').click(function(){
            $('.imagePicker').asImagePicker('val',{image: '../../images/gril.png'});
        });
        $('.test_val_empty').click(function(){
            $('.imagePicker').asImagePicker('val',null);
        });
        $('.test_get_val').click(function(){
            var val = $('.imagePicker').asImagePicker('val');
            console.info(val);
        });

        $('.test_enable').click(function(){
            $('.imagePicker').asImagePicker('enable');
        });
        $('.test_disable').click(function(){
            $('.imagePicker').asImagePicker('disable');
        });
        $('.test_destory').click(function(){
            $('.imagePicker').asImagePicker('destory');
        });

    });
</script>
