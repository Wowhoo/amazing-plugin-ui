Update

HTML

<div class="asRange7"></div>
<script>
    $(function(){
        $(".asRange7").asRange({
            namespace: 'asRange',
            step: 0.1,
            pointer: 1,
            min: 0
            max: 1
        });
        $(".asRange-update").on('click', function(){
            var instance = $('.asRange7').data('asRange');
            instance.update({
                step: 1,
                min: 8;
                max: 25;
                value: 10
            });
        });
    });
</script>
