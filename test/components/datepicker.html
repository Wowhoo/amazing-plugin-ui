Datepicker

Default

HTML

<input type="text" class="asCalendar asCalendar1"/>
<script>
    $(document).ready(function(){
        $(".asCalendar1").asDatepicker({
            namespace: 'asCalendar'
        });
        $('#asCalendar-api-show-click').click(function(){
            $('.asCalendar1').asDatepicker('show');
            return false;
        });
        $('#asCalendar-api-hide-click').click(function(){
            $('.asCalendar1').asDatepicker('hide');
            return false;
        });
        });
    });
</script>
